Ancestor path derivation algorithm (for filterPaths):
```typescript
// Input: matchingPaths = ['notes/hello.md', 'notes/sub/world.md']
// Output Set includes: 'notes/hello.md', 'notes', 'notes/sub/world.md', 'notes/sub', 'notes'
function addAncestors(paths: string[]): Set<string> {
  const result = new Set<string>();
  for (const p of paths) {
    result.add(p);
    const parts = p.split('/');
    for (let i = 1; i < parts.length; i++) {
      result.add(parts.slice(0, i).join('/'));
    }
  }
  return result;
}
```
