BacklinksPanel component spec (from RESEARCH.md):
```tsx
interface Props {
  activeFilePath: string | null;
  onOpen: (path: string) => void;
}

export function BacklinksPanel({ activeFilePath, onOpen }: Props) {
  const [backlinks, setBacklinks] = useState<string[]>([]);

  useEffect(() => {
    if (!activeFilePath) { setBacklinks([]); return; }
    fetch(`/api/backlinks/${activeFilePath}`)
      .then(r => r.json())
      .then((data: { backlinks: string[] }) => setBacklinks(data.backlinks))
      .catch(() => setBacklinks([]));
  }, [activeFilePath]);

  if (activeFilePath === null) return null;

  // Header always rendered: "Backlinks (N)" where N can be 0
  // Empty state: <p className="text-xs text-gray-300">No incoming links</p>
  // List items: button showing filename without extension, onClick calls onOpen(path)
  // Cap list height: max-h-40 overflow-y-auto (RESEARCH.md open question #2 recommendation)
  // Filename display: path.split('/').pop()?.replace(/\.md$/, '') ?? path
}
```

Styling guidance: Match FileInfo right-panel style — `px-4 py-3 border-b border-gray-200` wrapper, `text-sm font-semibold text-gray-700` for header, `text-xs text-orange-600 hover:text-orange-800` for clickable links. These are the classes used by the existing right-panel components.
